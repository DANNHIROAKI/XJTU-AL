;x、y、z 均为双精度数，分别存放在地址为X, X+2；Y, Y+2；Z, Z+2的存储单元中
;用指令序列实现w⬅x + y + 24 - z ，并用W, W+2单元存放w


; 加载32位双精度数x到AX和DX寄存器中
MOV AX, X     ; 将内存地址X处的16位值加载到AX寄存器
MOV DX, X+2   ; 将内存地址X+2处的16位值加载到DX寄存器
              ; 这样，AX和DX寄存器一起存储了32位整数x的值

; 执行x+y
ADD AX, Y     ; 将内存地址Y处的16位值与AX寄存器相加，并将结果存储在AX寄存器中
ADC DX, Y+2   ; 将内存地址Y+2处的16位值与DX寄存器相加，并将上一步产生的进位加入结果，将最终结果存储在DX寄存器中
              ; 此时，AX和DX寄存器一起存储了32位整数x + y的值

; 执行x+y+24
ADD AX, 24    ; 将立即数24与AX寄存器相加，并将结果存储在AX寄存器中
ADC DX, 0     ; 将立即数0与DX寄存器相加，并将上一步产生的进位加入结果，将最终结果存储在DX寄存器中
              ; 此时，AX和DX寄存器一起存储了32位整数x + y + 24的值

; 执行x+y+24-z
SUB AX, Z     ; 将内存地址Z处的16位值从AX寄存器中减去，并将结果存储在AX寄存器中
SBB DX, Z+2   ; 将内存地址Z+2处的16位值从DX寄存器中减去，并将上一步产生的借位从结果中减去，将结果存储在DX寄存器中
              ;AX和DX寄存器一起存储了32位整数x + y + 24 - z的值

; 将结果存储到内存地址W和W+2处
MOV W, AX     ; 将AX寄存器的16位值存储在内存地址W处
MOV W+2, DX   ; 将DX寄存器的16位值存储在内存地址W+2处
              ;此时，内存中的W和W+2单元一起存储了32位整数x + y + 24 - z的结果
